name: Stock Sentiment ML CI

on:
  push:
    branches:
      - main
  pull_request:
    branches:
      - main

jobs:
  train-and-validate:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.10"

      - name: Upgrade pip
        run: |
          python -m pip install --upgrade pip

      - name: Install dependencies
        run: |
          pip install -r requirements.txt

      - name: Download NLTK resources
        run: |
          python - <<EOF
          import nltk
          nltk.download('stopwords')
          nltk.download('wordnet')
          EOF

      - name: Run model training sanity check
        env:
          DATA_PATH: data/all-data.csv
        run: |
          python - <<EOF
          from sentiment_app import train_model
          train_model()
          EOF
